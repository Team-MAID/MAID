168 private bool IsHorizontalSplit()
169 {
170 // If width is 25% larger than height, we split vertically
171 // If height is 25% larger than width, we split horizontally
172 bool horizontal;
173 if (Boundary.width > Boundary.height && (float) Boundary.width / Boundary.height >= 1.25f)

183 horizontal = Random.Range(0f, 1f) > 0.5f;
